CREATE TABLE People
(
Id INT primary key identity,
Name NVARCHAR(200) NOT NULL,
Picture VARBINARY(MAX) CHECK(DATALENGTH(Picture) <= 2 * 1024 * 1024),
Height NUMERIC(3,2),
Weight NUMERIC(3,2),
Gender CHAR(1) CHECK([Gender] IN ('m','f')),
Birthdate DATE NOT NULL,
Biography NVARCHAR(MAX)
)
